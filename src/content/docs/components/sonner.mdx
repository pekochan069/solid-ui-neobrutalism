## Installation

<InstallationTabs name="sonner" code={SonnerCode}>
    Add the `Toaster` component to your root component (e.g., `App.tsx` or `index.tsx`)

```tsx ins={4,12}
// src/layout.tsx
import type { ParentProps } from "solid-js"

import { Toaster } from "~/components/ui/sonner"
 
export default function RootLayout(props: ParentProps) {
  return (
    <html lang="en">
      <head />
      <body>
        <main>{props.children}</main>
        <Toaster />
      </body>
    </html>
  )
}
```
